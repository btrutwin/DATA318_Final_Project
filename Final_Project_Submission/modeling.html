3. Helper — Evaluation Function

# Returns accuracy, AUC, and prints a confusion matrix for a given
# vector of predicted probabilities and true binary labels.
evaluate_model <- function(probs, truth, model_name, threshold = 0.5) {
  preds <- ifelse(probs >= threshold, 1, 0)
  acc   <- mean(preds == truth)
  roc_obj <- roc(truth, probs, quiet = TRUE)
  auc_val  <- as.numeric(auc(roc_obj))

  cat("\n===", model_name, "===\n")
  cat("Accuracy:", round(acc, 4), "\n")
  cat("AUC:     ", round(auc_val, 4), "\n")

  cm <- confusionMatrix(
    factor(preds, levels = c(0, 1), labels = c("Loss", "Win")),
    factor(truth,  levels = c(0, 1), labels = c("Loss", "Win")),
    positive = "Win"
  )
  print(cm$table)
  cat("Sensitivity (Recall):", round(cm$byClass["Sensitivity"], 4), "\n")
  cat("Specificity:         ", round(cm$byClass["Specificity"], 4), "\n")

  invisible(list(acc = acc, auc = auc_val, roc = roc_obj, cm = cm))
}
